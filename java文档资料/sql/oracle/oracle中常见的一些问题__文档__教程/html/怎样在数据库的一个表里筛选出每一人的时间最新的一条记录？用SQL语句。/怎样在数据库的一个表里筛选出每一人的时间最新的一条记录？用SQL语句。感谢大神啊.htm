/* 1 用子查询 */

select yhh, name, gdcs, gdsj1, gdtime

from table1

where exists

(

    select 1 from

    (

        select yhh, max(gdtime) as gdtime

        from table1 

        group by yhh

    ) x

    where x.yhh = table1.yhh 

      and x.gdtime = table1.gdtime

)



/* 2 用窗口函数 */

select * from

(

select yhh, name, gdcs, gdsj1, gdtime

, max(gdtime) over(partition by yhh) as gdtimeMax

from table1

) x

where gdtime = gdtimeMax
